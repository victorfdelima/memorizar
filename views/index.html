Criar conexão com banco de dados 

    
    const { Sequelize } = require('sequelize')
    
    const sequelize = new Sequelize('nomedobanco', 'root', 'password', {
        host: 'localhost',
        dialect: 'mysql', 
    })
    
    
    module.exports = sequelize

Dialect é a propriedade para usar o tipo de banco se é Mysql, mongodb ou outros
